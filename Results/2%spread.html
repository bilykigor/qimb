#Get error for these features bags
from sklearn.feature_selection import f_classif
from sklearn.metrics import confusion_matrix

error = []
nfeatures = range(1, X.shape[1])
labels = list(set(y))
for nfeature in nfeatures:
    X_r = X[fi['Feature'][:nfeature]]
    
    err = []
    for i in range(10):  
        r = range(len(dates))
        np.random.shuffle(r)
        test_days = r[:1] 
        train_days = r[1:] 
               
        Xtrain = X_r.ix[datesDF.ix[train_days],:]
        Xtest = X_r.ix[datesDF.ix[test_days],:]
        ytrain = y.ix[datesDF.ix[train_days]]
        ytest = y.ix[datesDF.ix[test_days]]
        
        reg = RF()
        
        reg.fit(Xtrain, ytrain)
        ypred = reg.predict(Xtest)
        
        test_cm = confusion_matrix(ytest,ypred,labels).astype(float)
        test_pr = qimbs.Precision_Recall(test_cm)
  
        err.append(1-test_pr[2])       
    error.append(np.mean(err))
    
plt.plot(nfeatures, error,'.')
plt.xlabel("n Important features selected")
plt.ylabel("Error")
plt.title('Features importance')

#Another method to estimate important features
from sklearn.metrics import confusion_matrix

error = []

nfeatures = range(1, X.shape[1])
labels = list(set(y))

for nfeature in nfeatures:
    # compute MSE for different values of k (top features)
    selector = SelectKBest(f_classif, k=nfeature)
    selector.fit(X, y)
    selected = selector.get_support()

    X_r = X[X.columns[selected]]
    
    err = []
    for i in range(5):  
        r = range(len(dates))
        np.random.shuffle(r)
        test_days = r[:1] 
        train_days = r[1:] 
               
        Xtrain = X_r.ix[datesDF.ix[train_days],:]
        Xtest = X_r.ix[datesDF.ix[test_days],:]
        ytrain = y.ix[datesDF.ix[train_days]]
        ytest = y.ix[datesDF.ix[test_days]]
        
        reg = SVC(class_weight='auto')
        
        reg.fit(Xtrain, ytrain)
        ypred = reg.predict(Xtest)
        
        test_cm = confusion_matrix(ytest,ypred,labels).astype(float)
        test_pr = qimbs.Precision_Recall(test_cm)
  
        err.append(1-test_pr[2])       
    error.append(np.mean(err))
    
plt.plot(nfeatures, error,'.')
plt.xlabel("Features selected")
plt.ylabel("Error")
plt.title('Features importance')
