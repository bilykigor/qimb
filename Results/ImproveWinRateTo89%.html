import pandas as pd
import matplotlib.pyplot as plt

#!Importing data
#!----------------
#!I import data into pandas dataframe.
import os.path
df = pd.read_csv('/home/user/PyProjects/data/2014-07-01/CommonAggr_2014-07-01.csv', low_memory=False)
for i in range (2,32):    
    if i<10:
        f = '/home/user/PyProjects/data/2014-07-0%s/CommonAggr_2014-07-0%s.csv' % (i,i)
    else:
        f = '/home/user/PyProjects/data/2014-07-%s/CommonAggr_2014-07-%s.csv' % (i,i)
    if (os.path.isfile(f)):
        df=df.append(pd.read_csv(f, low_memory=False))

target_cols = ['Date','Time','Fracs','Symbol','Reason','tSide','tPrice','tShares',
                'Bid_P', 'Bid_S', 'Ask_P', 'Ask_S', 
                'ImbRef','ImbCBC', 'ImbFar', 'ImbShares', 'ImbPaired']

df = df[target_cols]

dates = list(set(df.Date))
df.index = range(df.shape[0])

#!I prefare to work with one variable that represents datetime. So I combine Date,Time,Fracs into one string (I don't need nanoseconds and thus I take only 0-2,4-6 chars in Fracs).
#!Then I parse this string with datetime.strptime and append the result to Timestamp list.
Timestamp = []
for i in range(df.shape[0]):
    Timestamp.append(datetime.strptime(df.Date[i] +' '+df.Time[i]+' '+df.Fracs[i][0:3]+df.Fracs[i][4:7],'%Y-%m-%d %H:%M:%S %f'))

#!I include the new created Timestamp feature into df dataframe. Now all necessary datetime data is in Timestamp and we drop  Date,Time,Fracs since they are no more needed.  
df['Timestamp'] = Timestamp
df = df.set_index(['Timestamp'])
df = df.drop(['Date','Time','Fracs'],1)

def sigmoid(x):
    return 1/(1+exp(-x))

fdf = pd.DataFrame()
fdf['Pnl'] = \
100*(Merged.tPrice - Merged.Ask_P)*(Merged.ImbShares>0)/Merged.Ask_P + \
100*(Merged.Bid_P - Merged.tPrice)*(Merged.ImbShares<0)/Merged.Bid_P 
fdf['RefDist'] = sigmoid(200/(Merged.Ask_P + Merged.Bid_P)* \
((Merged.ImbShares<0)*(Merged.Bid_P-Merged.ImbRef) + \
(Merged.ImbShares>0)*(Merged.ImbRef-Merged.Ask_P)))
fdf['Spread'] = 2*(Merged.Ask_P - Merged.Bid_P)/(Merged.Ask_P + Merged.Bid_P)
fdf['ABSN'] = sigmoid((Merged.Ask_S - Merged.Bid_S)/Merged.ImbShares)
fdf['Imb'] = sigmoid(Merged.ImbShares/Merged.tShares)
fdf['ImbNotNormed'] = Merged.ImbShares/Merged.tShares

from sklearn import preprocessing
fdf['ScaledSpread'] = preprocessing.scale(fdf['Spread'])
